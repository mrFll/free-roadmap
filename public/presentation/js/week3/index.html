Example

function getMax(x, y) {
    var result;

    if (x > y) {
        result = x;
    } else {
        result = y;
    }

    return result;
}

function getMaxFour(x, y, z, w) {
    var max1 = getMax(x, y);
    var max2 = getMax(z, w);
    var result = getMax(max1, max2);
    return result;
}
